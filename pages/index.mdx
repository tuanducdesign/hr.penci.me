import React, { useMemo } from 'react'
import Link from '@components/Link'
import Hero from '@components/home/Hero'
import { getPagesUnderRoute } from 'nextra/context'
import { BsLayoutSplit, BsServer, BsPaletteFill } from 'react-icons/bs'

export function DocsFrontEnd() {
  const MAX_DISPLAY = 6
  const pages = useMemo(() => {
    return getPagesUnderRoute('/docs/front-end').filter((page, index) => index < MAX_DISPLAY)
  }, [])
  return (
    <div className="mt-10">
      <div className="space-y-4 sm:grid sm:grid-cols-2 lg:grid-cols-3 sm:gap-4 xl:gap-8 sm:space-y-0">
        {pages.map((page) => {
          return (
            <Link
              key={page.route}
              href={page.route}
              title={page.meta?.title || page.frontMatter?.title || page.name}
              className="block py-12 px-8 text-center rounded-lg bg-white dark:bg-[#202020]"
            >
              <h3 className="font-semibold text-xl">
                {page.meta?.title || page.frontMatter?.title || page.name}
              </h3>
            </Link>
          )
        })}
      </div>
    </div>
  )
}
